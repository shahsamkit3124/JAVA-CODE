class Q11
{
    public static void main(String args[])
    {
        Scanner sc = new Scanner(System.in);
        int x = sc.nextInt(), rem=0, sum=0;
        int t=x;
        while(x>0)
        {
            rem = x%10;
            sum += Math.pow(rem,3);
            x = x/10;
        }
        System.out.println(sum==t?"Armstrong":"Not");
    }
}
