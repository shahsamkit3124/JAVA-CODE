class Q13
{
    public static void main(String ar[])
    {
        Scanner sc = new Scanner(System.in);
        int x = sc.nextInt(), rem=0, rev=0;
        while(x>0)
        {
            rem = x%10;
            rev = rev*10 + rem;
            x = x/10;
        }
        while(rev>0)
        {
            rem = rev%10;
            System.out.print(rem+" ");
            rev/=10;
        }
    }
}
